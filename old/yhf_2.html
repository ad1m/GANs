Previously, we scraped some adj close prices for a list of tickers. Assuming that these tickers are in our portfolio, let's calculate portfolio value. Our function will take in the adjusted close prices dataframe, the weights of each stock in our portfolio, and our initial cash.

We need to normalize all our adjusted close prices. We can do so by dividing each price by the first price in the dataframe (index 0). We then need to calculate our allocations, to do this we take our normalized dataframe and multiply it by the weights. We can then calculate our portfolio values by multiplying our allocations dataframe by our cash. We then sum the columns by using the function .sum(axis=1).

import pandas as pd
import pandas_datareader.data as web
def get_adjusted_close(tickers,start,end): 
    d = {} #our dictionary 
    for ticker in tickers: 
        d[ticker] = web.DataReader(ticker,"yahoo",start,end) #this will give us open, high, low, close, volume, Adj Close
    pan = pd.Panel(d)
    df_adj_close = pan.minor_xs('Adj Close')
    return df_adj_close

def Portfolio_Value(df_adj_close,weights,cash):
        normed = df_adj_close/df_adj_close.ix[0,:] #Norming the prices
        alloced = normed*weights #Creating our allocations
        pos_vals = alloced*cash  #Our portfolio value 
        series_portfolio_value = pos_vals.sum(axis=1)  #Summing by row 
        return series_portfolio_value

    
tickers = ['AAPL','BAC','GILD','MSFT']
start = '2016-01-01'
end = '2016-12-21'
df_adj_close = get_adjusted_close(tickers,start,end)
weights = [0.25,0.25,0.25,0.25]
cash = 100000
pv = Portfolio_Value(df_adj_close,weights,cash)

def Plot_Portfolio_Value(port_val):
        plt.style.use('ggplot')   #The style of our plot 
        
        port_val = port_val.to_frame()
        dates =  port_val.index
        port_val = port_val.values
        fig = plt.figure()
        ax = fig.add_subplot(111)
        fig.subplots_adjust(top=0.85)
        
        #Setting our axis titles 
        ax.set_title('Portfolio Value') 
        ax.set_xlabel('Date')
        ax.set_ylabel('Portfolio Value ($)')
        
        #plotting our portfolio
        plt.plot(dates,port_val)
        return
